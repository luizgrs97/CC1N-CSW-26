Neste exemplo vou demonstrar um trecho de código

Inserindo uma rotina em batch script:

        
    :ro_number
    set "isnumber=" & set "num="
    set "tempstr=%~1"
    for %%Z in (%numbers%) do (
        if "!tempstr!" neq "" set "tempstr=!tempstr:%%Z=!"
    )
    if "!tempstr!" neq "" exit /b
    set "num=%~1"
    set "num=!num:0=!"
    if not "%~1"=="" if not "!num!"=="" (
        for /f %%Z in ("!num:~0,1!") do (
            set "num=%~1"
            set num=!num:*%%Z=%%Z!
            set isnumber=1
        )
    )
    exit /b
